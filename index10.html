using System; class Program { static void Main() { string[] strings = { "apple", "banana", "orange", "grape", "pear" }; Console.Write("Введите строку: "); string input = Console.ReadLine(); bool found = false; foreach (string str in strings) { if (str.Equals(input, StringComparison.OrdinalIgnoreCase)) { found = true; break; } } if (found) { Console.WriteLine("Строка найдена!"); } else { Console.WriteLine("Строка не найдена!"); } } }
